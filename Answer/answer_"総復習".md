09.5つの文字型の配列をカンマ区切りで表示しましょう(拡張for文を使用)
```java
public class Main {
	public static void main(String[] args) {

		String[] names = { "名前1", "名前2", "名前3", "名前4", "名前5" };

		int i = 0;
		for( String name : names ){
			if( i<names.length-1 ){
				name += ",";
			}
			System.out.print( name );
			i++;
		}

	}
}
```
